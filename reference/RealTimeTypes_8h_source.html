43  /*
44  RTTI (Run-Time Type Identification) is special compiler-generated information which is
45  used to support C++ operators like dynamic_cast<> and typeid(), and also for C++ exceptions.
46  Due to its nature, RTTI is only required (and generated) for polymorphic classes,
47  i.e. classes with virtual functions.
48 
49  MSVC compiler puts a pointer to the structure called "Complete Object Locator" just before
50  the vftable. The structure is called so because it allows compiler to find the location of
51  the complete object from a specific vftable pointer (since a class can have several of them).
52  */
53  struct RTTICompleteObjectLocator
54  {
55  DWORD signature; //always zero ?
56  DWORD offset; //offset of this vtable in the complete class
57  DWORD cdOffset; //constructor displacement offset
58  ti_ptr pTypeDescriptor; //TypeDescriptor of the complete class
59  RTTIClassHierarchyDescriptor_ptr pClassDescriptor; //describes inheritance hierarchy
60  };
