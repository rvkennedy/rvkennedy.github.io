226  void EnsureCubemapRenderTargets(crossplatform::RenderPlatform *renderPlatform,void *t)
227  {
228  ID3D11Texture2D *tex=(ID3D11Texture2D *)t;
229  if(tex==renderTargetTexture2D&&cubemapRenderTargets[0]!=NULL)
230  return;
231  for(int i=0;i<6;i++)
232  SAFE_RELEASE(cubemapRenderTargets[i]);
233 
234  renderTargetTexture2D=tex;
235  if(!renderTargetTexture2D)
236  return;
237 #ifdef _MSC_VER
238  D3D11_TEXTURE2D_DESC desc;
239  renderTargetTexture2D->GetDesc(&desc);
240  if(desc.ArraySize==6)
241  {
242  // Create the multi-face render target view
243  D3D11_RENDER_TARGET_VIEW_DESC DescRT;
244  DescRT.Format =simul::dx11::TypelessToSrvFormat(desc.Format);//;
245  DescRT.ViewDimension =D3D11_RTV_DIMENSION_TEXTURE2DARRAY;
246  DescRT.Texture2DArray.FirstArraySlice =0;
247  DescRT.Texture2DArray.ArraySize =6;
248  DescRT.Texture2DArray.MipSlice =0;
249  for(int i=0;i<6;i++)
250  {
251  DescRT.Texture2DArray.FirstArraySlice = i;
252  DescRT.Texture2DArray.ArraySize = 1;
253  HRESULT hr=(renderPlatform->AsD3D11Device()->CreateRenderTargetView(tex, &DescRT, &(cubemapRenderTargets[i])));
254  if (hr != S_OK)
255  {
256  SIMUL_CERR<< "Error creating Cubemap rendertarget view." << std::endl;
257  }
258  }
259  }
260 #endif
261  }

374 UE4RENDERINTERFACE_EXPORT void StaticSetGraphicsDevice(void* device, GfxDeviceRenderer deviceType, GfxDeviceEventType eventType)
375 {
376  // This might happen before Init() so create readWriteMutex in that case!
377  if(!readWriteMutex)
378  readWriteMutex=new simul::base::ReadWriteMutex;
379  readWriteMutex->lock_for_write();
380  if(eventType==kGfxDeviceEventInitialize)
381  {
382  // Set device type to -1, i.e. "not recognized by our plugin"
383  g_DeviceType = kGfxRendererUnknown;
384 
385  #if SUPPORT_D3D11
386  // D3D11 device, remember device pointer and device type.
387  // The pointer we get is ID3D11Device.
388  if (deviceType == kGfxRendererD3D11)
389  {
390  if (eventType == kGfxDeviceEventInitialize)
391  unityDeviceValues.pd3dDevice=(ID3D11Device*)device;
392  g_DeviceType = deviceType;
393  unityDeviceValues.changed=true;
394  }
395  #endif
396 
397  if (deviceType == kGfxRendererOpenGL)
398  {
399  DebugLog ("trueSKY currently supports only DirectX 11.\n");
400  g_DeviceType = deviceType;
401  }
402  #ifdef __ORBIS__
403  if (deviceType == kGfxRendererPS4)
404  {
405  //SIMUL_CERR<<"StaticSetGraphicsDevice: device is "<<std::hex<<device<<std::dec<<std::endl;
406  if (eventType == kGfxDeviceEventInitialize)
407  unityDeviceValues.pd3dDevice=(ID3D11Device*)device;
408  // renderPlatformPS4.GetImmediateContext((sce::Gnmx::LightweightGfxContext*)device);
409  g_DeviceType = deviceType;
410  //if(device)
411 // ((sce::Gnmx::LightweightGfxContext*)device)->validate();
412  }
413  #endif
414  AddPaths();
415  }
416  readWriteMutex->unlock_from_write();
417  if(eventType==kGfxDeviceEventShutdown)
418  {
419  StaticShutDownInterface();
420  if(currentRenderPlatform)
421  currentRenderPlatform->InvalidateDeviceObjects();
422  #ifdef __ORBIS__
423  if (deviceType == kGfxRendererPS4)
424  {
425  StaticOnDeviceChanged(NULL);
426  }
427  #endif
428  if (deviceType == kGfxRendererD3D11)
429  {
430  // Might not get another render event so shut down now:
431  unityDeviceValues.pd3dDevice=NULL;
432  StaticOnDeviceChanged(unityDeviceValues.pd3dDevice);
433  }
434  // Set device type to -1, i.e. "not recognized by our plugin"
435  g_DeviceType = kGfxRendererUnknown;
436  }
437 }
