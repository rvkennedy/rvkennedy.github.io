316 TRUESKYRENDERINTERFACE_EXPORT void simul::plugin::StaticSetGraphicsDevice(void* device, GfxDeviceRenderer deviceType, GfxDeviceEventType eventType)
317 {
318  // This might happen before Init() so create readWriteMutex in that case!
319  if(!readWriteMutex)
320  readWriteMutex=new simul::base::ReadWriteMutex;
321  readWriteMutex->lock_for_write();
322  if(eventType==kGfxDeviceEventInitialize)
323  {
324  // Set device type to -1, i.e. "not recognized by our plugin"
325  g_DeviceType = kGfxRendererUnknown;
326 
327  #if SUPPORT_D3D11
328  // D3D11 device, remember device pointer and device type.
329  // The pointer we get is ID3D11Device.
330  if (deviceType == kGfxRendererD3D11)
331  {
332  if (eventType == kGfxDeviceEventInitialize)
333  unityDeviceValues.pd3dDevice=(ID3D11Device*)device;
334  g_DeviceType = deviceType;
335  unityDeviceValues.changed=true;
336  }
337  #endif
338 
339  if (deviceType == kGfxRendererOpenGL)
340  {
341  DebugLog ("trueSKY currently supports only DirectX 11.\n");
342  g_DeviceType = deviceType;
343  }
344  #ifdef __ORBIS__
345  if (deviceType == kGfxRendererPS4)
346  {
347  unityDeviceValues.pd3dDevice=(ID3D11Device*)device;
348  g_DeviceType = deviceType;
349  }
350  #endif
351  AddPaths();
352  }
353  readWriteMutex->unlock_from_write();
354  if(eventType==kGfxDeviceEventShutdown)
355  {
356  StaticShutDownInterface();
357  if(currentRenderPlatform)
358  currentRenderPlatform->InvalidateDeviceObjects();
359  #ifdef __ORBIS__
360  if (deviceType == kGfxRendererPS4)
361  {
362  InitDevice(NULL);
363  }
364  #endif
365  if (deviceType == kGfxRendererD3D11)
366  {
367  // Might not get another render event so shut down now:
368  unityDeviceValues.pd3dDevice=NULL;
369  }
370  // Set device type to -1, i.e. "not recognized by our plugin"
371  g_DeviceType = kGfxRendererUnknown;
372  }
