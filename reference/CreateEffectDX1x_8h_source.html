83 inline void D3DXMatrixInverse(D3DXMATRIX *dest,D3DXVECTOR3 *det,const D3DXMATRIX *src)
84 {
85  DirectX::XMMATRIX d;
86  DirectX::XMMATRIX s=DirectX::XMLoadFloat4x4(src);
87  if(det)
88  {
89  DirectX::XMVECTOR det2=DirectX::XMLoadFloat3(det);
90  d=DirectX::XMMatrixInverse(&det2,s);
91  DirectX::XMStoreFloat3(det,det2);
92  }
93  else
94  {
95  d=DirectX::XMMatrixInverse(NULL,s);
96  }
97  DirectX::XMStoreFloat4x4(dest,d);
98 }
