105  inline vec4 GetTransformLowResToHiRes() const
106  {
107  float uu=(downscale>1)?0.5f:0.0f;
108  float A=(float)(w*downscale)/(float)W;
109  float B=(float)(h*downscale)/(float)H;
110  float X=(float)(-pixelOffset.x)/(float)W-uu*(1.f/w)*A;
111  float Y=(float)(-pixelOffset.y)/(float)H-uu*(1.f/h)*B;
112  return vec4(X,Y,A,B);
113  }
114  // X = -px/W - uu/w*a
115  // U = X + A u
116  // =>
117  // u= (U-X)/A
118  // =- X/A+(1/A)U

121 
122  // where x=-X/A and a=1/A
123  inline vec4 GetTransformHiResToLowRes() const
124  {
125  if(downscale<=1)
